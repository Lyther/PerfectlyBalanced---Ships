namespace = pbships_start_menu

# game start event
event = {
	id = pbships_start_menu.0
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = pbships_initialized }
		is_ai = no
	}

	immediate = {
		set_global_flag = pbships_initialized # set global identifier
		set_global_flag = pbships_draw_low # default option
		country_event = { id = pbships_start_menu.1 }
	}
}
